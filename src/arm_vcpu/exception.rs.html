52/// Handles synchronous exceptions that occur during the execution of a guest VM.
53///
54/// This function examines the exception class (EC) to determine the cause of the exception
55/// and then handles it accordingly.
56///
57/// Currently we just handle exception type including data abort (`DataAbortLowerEL`) and hypervisor call (`HVC64)`.
58///
59/// # Arguments
60///
61/// * `ctx` - A mutable reference to the `TrapFrame`, which contains the saved state of the guest VM's CPU registers at the time of the exception.
62///
63/// # Returns
64///
65/// An `AxResult` containing an `AxVCpuExitReason` indicating the reason for the VM exit.
66/// This could be due to a hypervisor call (`Hypercall`) or other reasons such as data aborts.
67///
68/// # Panics
69///
70/// If an unhandled exception class is encountered, the function will panic, outputting
71/// details about the exception including the instruction pointer, faulting address, exception
72/// syndrome register (ESR), and system control registers.
73///
74pub fn handle_exception_sync(ctx: &mut TrapFrame) -> AxResult<AxVCpuExitReason> {
75    match exception_class() {
76        Some(ESR_EL2::EC::Value::DataAbortLowerEL) => {
77            let elr = ctx.exception_pc();
78            let val = elr + exception_next_instruction_step();
79            ctx.set_exception_pc(val);
80            handle_data_abort(ctx)
81        }
82        Some(ESR_EL2::EC::Value::HVC64) => {
83            // The `#imm`` argument when triggering a hvc call, currently not used.
84            let _hvc_arg_imm16 = ESR_EL2.read(ESR_EL2::ISS);
85
86            // Is this a psci call?
87            //
88            // By convention, a psci call can use either the `hvc` or the `smc` instruction.
89            // NimbOS uses `hvc`, `ArceOS` use `hvc` too when running on QEMU.
90            if let Some(result) = handle_psci_call(ctx) {
91                return result;
92            }
93
94            // We assume that guest VM triggers HVC through a `hvc #0`` instruction.
95            // And arm64 hcall implementation uses `x0` to specify the hcall number.
96            // For more details on the hypervisor call (HVC) mechanism and the use of general-purpose registers,
97            // refer to the [Linux Kernel documentation on KVM ARM hypervisor ABI](https://github.com/torvalds/linux/blob/master/Documentation/virt/kvm/arm/hyp-abi.rst).
98            Ok(AxVCpuExitReason::Hypercall {
99                nr: ctx.gpr[0],
100                args: [
101                    ctx.gpr[1], ctx.gpr[2], ctx.gpr[3], ctx.gpr[4], ctx.gpr[5], ctx.gpr[6],
102                ],
103            })
104        }
105        Some(ESR_EL2::EC::Value::TrappedMsrMrs) => handle_system_register(ctx),
106        Some(ESR_EL2::EC::Value::SMC64) => {
107            let elr = ctx.exception_pc();
108            let val = elr + exception_next_instruction_step();
109            ctx.set_exception_pc(val);
110            handle_smc64_exception(ctx)
111        }
112        _ => {
113            panic!(
114                "handler not presents for EC_{} @ipa 0x{:x}, @pc 0x{:x}, @esr 0x{:x},
115                @sctlr_el1 0x{:x}, @vttbr_el2 0x{:x}, @vtcr_el2: {:#x} hcr: {:#x} ctx:{}",
116                exception_class_value(),
117                exception_fault_addr()?,
118                (*ctx).exception_pc(),
119                exception_esr(),
120                SCTLR_EL1.get() as usize,
121                VTTBR_EL2.get() as usize,
122                VTCR_EL2.get() as usize,
123                HCR_EL2.get() as usize,
124                ctx
125            );
126        }
127    }
128}

130fn handle_data_abort(context_frame: &mut TrapFrame) -> AxResult<AxVCpuExitReason> {
131    let addr = exception_fault_addr()?;
132    let access_width = exception_data_abort_access_width();
133    let is_write = exception_data_abort_access_is_write();
134    //let sign_ext = exception_data_abort_access_is_sign_ext();
135    let reg = exception_data_abort_access_reg();
136    let reg_width = exception_data_abort_access_reg_width();
137
138    trace!(
139        "Data fault @{:?}, ELR {:#x}, esr: 0x{:x}",
140        addr,
141        context_frame.exception_pc(),
142        exception_esr(),
143    );
144
145    let width = match AccessWidth::try_from(access_width) {
146        Ok(access_width) => access_width,
147        Err(_) => return Err(AxError::InvalidInput),
148    };
149
150    let reg_width = match AccessWidth::try_from(reg_width) {
151        Ok(reg_width) => reg_width,
152        Err(_) => return Err(AxError::InvalidInput),
153    };
154
155    if !exception_data_abort_handleable() {
156        panic!(
157            "Core data abort not handleable {:#x}, esr {:#x}",
158            addr,
159            exception_esr()
160        );
161    }
162
163    if !exception_data_abort_is_translate_fault() {
164        if exception_data_abort_is_permission_fault() {
165            return Err(AxError::Unsupported);
166        } else {
167            panic!("Core data abort is not translate fault {:#x}", addr,);
168        }
169    }
170
171    if is_write {
172        return Ok(AxVCpuExitReason::MmioWrite {
173            addr,
174            width,
175            data: context_frame.gpr(reg) as u64,
176        });
177    }
178    Ok(AxVCpuExitReason::MmioRead {
179        addr,
180        width,
181        reg,
182        reg_width,
183        signed_ext: false,
184    })
185}

187/// Handles a system register access exception.
188///
189/// This function processes the exception by reading or writing to a system register
190/// based on the information in the `context_frame`.
191///
192/// # Arguments
193/// * `context_frame` - A mutable reference to the trap frame containing the CPU state.
194///
195/// # Returns
196/// * `AxResult<AxVCpuExitReason>` - An `AxResult` containing an `AxVCpuExitReason` indicating
197///   whether the operation was a read or write and the relevant details.
198fn handle_system_register(context_frame: &mut TrapFrame) -> AxResult<AxVCpuExitReason> {
199    let iss = ESR_EL2.read(ESR_EL2::ISS);
200
201    let addr = exception_sysreg_addr(iss.try_into().unwrap());
202    let elr = context_frame.exception_pc();
203    let val = elr + exception_next_instruction_step();
204    let write = exception_sysreg_direction_write(iss);
205    let reg = exception_sysreg_gpr(iss) as usize;
206    context_frame.set_exception_pc(val);
207    if write {
208        return Ok(AxVCpuExitReason::SysRegWrite {
209            addr: SysRegAddr::new(addr),
210            value: context_frame.gpr(reg as usize) as u64,
211        });
212    }
213    Ok(AxVCpuExitReason::SysRegRead {
214        addr: SysRegAddr::new(addr),
215        reg,
216    })
217}

219/// Handles HVC or SMC exceptions that serve as psci (Power State Coordination Interface) calls.
220///
221/// A hvc or smc call with the function in range 0x8000_0000..=0x8000_001F  (when the 32-bit
222/// hvc/smc calling convention is used) or 0xC000_0000..=0xC000_001F (when the 64-bit hvc/smc
223/// calling convention is used) is a psci call. This function handles them all.
224///
225/// Returns `None` if the HVC is not a psci call.
226fn handle_psci_call(ctx: &mut TrapFrame) -> Option<AxResult<AxVCpuExitReason>> {
227    const PSCI_FN_RANGE_32: core::ops::RangeInclusive<u64> = 0x8400_0000..=0x8400_001F;
228    const PSCI_FN_RANGE_64: core::ops::RangeInclusive<u64> = 0xC400_0000..=0xC400_001F;
229
230    const PSCI_FN_VERSION: u64 = 0x0;
231    const _PSCI_FN_CPU_SUSPEND: u64 = 0x1;
232    const PSCI_FN_CPU_OFF: u64 = 0x2;
233    const PSCI_FN_CPU_ON: u64 = 0x3;
234    const _PSCI_FN_MIGRATE: u64 = 0x5;
235    const PSCI_FN_SYSTEM_OFF: u64 = 0x8;
236    const _PSCI_FN_SYSTEM_RESET: u64 = 0x9;
237    const PSCI_FN_END: u64 = 0x1f;
238
239    let fn_ = ctx.gpr[0];
240    let fn_offset = if PSCI_FN_RANGE_32.contains(&fn_) {
241        Some(fn_ - PSCI_FN_RANGE_32.start())
242    } else if PSCI_FN_RANGE_64.contains(&fn_) {
243        Some(fn_ - PSCI_FN_RANGE_64.start())
244    } else {
245        None
246    };
247
248    match fn_offset {
249        Some(PSCI_FN_CPU_OFF) => Some(Ok(AxVCpuExitReason::CpuDown { _state: ctx.gpr[1] })),
250        Some(PSCI_FN_CPU_ON) => Some(Ok(AxVCpuExitReason::CpuUp {
251            target_cpu: ctx.gpr[1],
252            entry_point: GuestPhysAddr::from(ctx.gpr[2] as usize),
253            arg: ctx.gpr[3],
254        })),
255        Some(PSCI_FN_SYSTEM_OFF) => Some(Ok(AxVCpuExitReason::SystemDown)),
256        // We just forward these request to the ATF directly.
257        Some(PSCI_FN_VERSION..PSCI_FN_END) => None,
258        _ => None,
259    }
260}

262/// Handles SMC (Secure Monitor Call) exceptions.
263///
264/// This function will judge if the SMC call is a PSCI call, if so, it will handle it as a PSCI call.
265/// Otherwise, it will forward the SMC call to the ATF directly.
266fn handle_smc64_exception(ctx: &mut TrapFrame) -> AxResult<AxVCpuExitReason> {
267    // Is this a psci call?
268    if let Some(result) = handle_psci_call(ctx) {
269        result
270    } else {
271        // We just forward the SMC call to the ATF directly.
272        // The args are from lower EL, so it is safe to call the ATF.
273        (ctx.gpr[0], ctx.gpr[1], ctx.gpr[2], ctx.gpr[3]) =
274            unsafe { crate::smc::smc_call(ctx.gpr[0], ctx.gpr[1], ctx.gpr[2], ctx.gpr[3]) };
275        Ok(AxVCpuExitReason::Nothing)
276    }
277}

294    let iss = ESR_EL2.read(ESR_EL2::ISS);

297    error!("Exception Class: {ec:#x}");

301        "Unhandled synchronous exception from current EL: {:#x?}",
302        tf

306/// A trampoline function for sp switching during handling VM exits,
307/// when **there is a active VCPU running**, which means that the host context is stored
308/// into host stack in `run_guest` function.
309///
310/// # Functionality
311///
312/// 1. **Restore Previous Host Stack pointor:**
313///     - The guest context frame is aleady saved by `SAVE_REGS_FROM_EL1` macro in exception.S.
314///       This function firstly adjusts the `sp` to skip the exception frame
315///       (adding `34 * 8` to the stack pointer) according to the memory layout of `Aarch64VCpu` struct,
316///       which makes current `sp` point to the address of `host_stack_top`.
317///       The host stack top value is restored by `ldr`.
318///
319/// 2. **Restore Host Context:**
320///     - The `restore_regs_from_stack!()` macro is invoked to restore the host function context
321///       from the stack. This macro handles the restoration of the host's callee-saved general-purpose
322///       registers (`x19` to `x30`).
323///
324/// 3. **Restore Host Control Flow:**
325///     - The `ret` instruction is used to return control to the host context after
326///       the guest context has been saved in `Aarch64VCpu` struct and the host context restored.
327///       Finally the control flow is returned back to `Aarch64VCpu.run()` in [vcpu.rs].
328///
329/// # Notes
330///
331/// - This function is typically invoked when a VM exit occurs, requiring the
332///   hypervisor to switch context from the guest to the host. The precise control
333///   over stack and register management ensures that the transition is smooth and
334///   that the host can correctly resume execution.
335///
336/// - The `options(noreturn)` directive indicates that this function will not return
337///   to its caller, as control will be transferred back to the host context via `ret`.
338///
339/// - This function is not typically called directly from Rust code. Instead, it is
340///   invoked as part of the low-level hypervisor or VM exit handling routines.
341#[unsafe(naked)]
342#[unsafe(no_mangle)]
343unsafe extern "C" fn vmexit_trampoline() -> ! {
344    core::arch::naked_asm!(
345        // Curretly `sp` points to the base address of `Aarch64VCpu.ctx`, which stores guest's `TrapFrame`.
346        "add x9, sp, 34 * 8", // Skip the exception frame.
347        // Currently `x9` points to `&Aarch64VCpu.host_stack_top`, see `run_guest()` in vcpu.rs.
348        "ldr x10, [x9]", // Get `host_stack_top` value from `&Aarch64VCpu.host_stack_top`.
349        "mov sp, x10",   // Set `sp` as the host stack top.
350        restore_regs_from_stack!(), // Restore host function context frame.
351        "ret", // Control flow is handed back to Aarch64VCpu.run(), simulating the normal return of the `run_guest` function.
352    )
353}
