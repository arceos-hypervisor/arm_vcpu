1use core::{cell::OnceCell, marker::PhantomData};
2
3use aarch64_cpu::registers::*;
4use axerrno::AxResult;
5use axvcpu::{AxArchPerCpu, AxVCpuHal};
6use tock_registers::interfaces::ReadWriteable;
7
8/// Per-CPU data. A pointer to this struct is loaded into TP when a CPU starts. This structure
9#[repr(C)]
10#[repr(align(4096))]
11pub struct Aarch64PerCpu<H: AxVCpuHal> {
12    /// per cpu id
13    pub cpu_id: usize,
14    _phantom: PhantomData<H>,
15}
16
17#[percpu::def_percpu]
18static ORI_EXCEPTION_VECTOR_BASE: usize = 0;
19
20/// IRQ handler registered by underlying host OS during per-cpu initialization,
21/// for dispatching IRQs to the host OS.
22///
23/// Set `IRQ_HANDLER` as per-cpu variable to avoid the need of `OnceLock`.
24#[percpu::def_percpu]
25pub static IRQ_HANDLER: OnceCell<&(dyn Fn() + Send + Sync)> = OnceCell::new();
26
27unsafe extern "C" {
28    fn exception_vector_base_vcpu();
29}
30
31impl<H: AxVCpuHal> AxArchPerCpu for Aarch64PerCpu<H> {
32    fn new(cpu_id: usize) -> AxResult<Self> {
33        // Register IRQ handler for this CPU.
34        let _ = unsafe { IRQ_HANDLER.current_ref_mut_raw() }
35            .set(&|| H::irq_hanlder())
36            .map(|_| {});
37
38        Ok(Self {
39            cpu_id,
40            _phantom: PhantomData,
41        })
42    }
43
44    fn is_enabled(&self) -> bool {
45        HCR_EL2.is_set(HCR_EL2::VM)
46    }
47
48    fn hardware_enable(&mut self) -> AxResult {
49        // First we save origin `exception_vector_base`.
50        // Safety:
51        // Todo: take care of `preemption`
52        unsafe { ORI_EXCEPTION_VECTOR_BASE.write_current_raw(VBAR_EL2.get() as usize) }
53
54        // Set current `VBAR_EL2` to `exception_vector_base_vcpu`
55        // defined in this crate.
56        VBAR_EL2.set(exception_vector_base_vcpu as usize as _);
57
58        HCR_EL2.modify(
59            HCR_EL2::VM::Enable
60                + HCR_EL2::RW::EL1IsAarch64
61                + HCR_EL2::IMO::EnableVirtualIRQ
62                + HCR_EL2::FMO::EnableVirtualFIQ
63                + HCR_EL2::TSC::EnableTrapEl1SmcToEl2,
64        );
65
66        // Note that `ICH_HCR_EL2` is not the same as `HCR_EL2`.
67        //
68        // `ICH_HCR_EL2[0]` controls the virtual CPU interface operation.
69        //
70        // We leave it for the virtual GIC implementations to decide whether to enable it or not.
71        //
72        // unsafe {
73        //     core::arch::asm! {
74        //         "msr ich_hcr_el2, {value:x}",
75        //         value = in(reg) 0,
76        //     }
77        // }
78
79        Ok(())
80    }
81
82    fn hardware_disable(&mut self) -> AxResult {
83        // Reset `VBAR_EL2` into previous value.
84        // Safety:
85        // Todo: take care of `preemption`
86        VBAR_EL2.set(unsafe { ORI_EXCEPTION_VECTOR_BASE.read_current_raw() } as _);
87
88        HCR_EL2.set(HCR_EL2::VM::Disable.into());
89        Ok(())
90    }
91}
