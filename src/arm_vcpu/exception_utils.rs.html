33/// Retrieves the Hypervisor IPA Fault Address Register (HPFAR) value from EL2.
34///
35/// This function uses inline assembly to read the HPFAR_EL2 register.
36///
37/// # Returns
38/// The value of the HPFAR_EL2 register as a `usize`.
39#[inline(always)]
40fn exception_hpfar() -> usize {
41    let hpfar: u64;
42    unsafe {
43        core::arch::asm!("mrs {}, HPFAR_EL2", out(reg) hpfar);
44    }
45    hpfar as usize
46}

48/// Constant for the shift amount used to identify the S1PTW bit in ESR_ELx.
49#[allow(non_upper_case_globals)]
50const ESR_ELx_S1PTW_SHIFT: usize = 7;
51/// Constant representing the S1PTW (Stage 1 translation fault) bit in ESR_ELx.
52#[allow(non_upper_case_globals)]
53const ESR_ELx_S1PTW: usize = 1 << ESR_ELx_S1PTW_SHIFT;

55/// Macro for executing an ARM Address Translation (AT) instruction.
56///
57/// The macro takes two arguments:
58/// - `$at_op`: The AT operation to perform (e.g., `"s1e1r"`).
59/// - `$addr`: The address on which to perform the AT operation.
60///
61/// This macro is unsafe because it directly executes assembly code.
62///
63/// Example usage:
64/// ```rust
65/// arm_at!("s1e1r", address);
66/// ```
67macro_rules! arm_at {
68    ($at_op:expr, $addr:expr) => {
69        unsafe {
70            core::arch::asm!(concat!("AT ", $at_op, ", {0}"), in(reg) $addr, options(nomem, nostack));
71            core::arch::asm!("isb");
72        }
73    };
74}

76/// Translates a Fault Address Register (FAR) to a Hypervisor Physical Fault Address Register (HPFAR).
77///
78/// This function uses the ARM Address Translation (AT) instruction to translate
79/// the provided FAR to an HPFAR. The translation result is returned in the Physical
80/// Address Register (PAR_EL1), and is then converted to the HPFAR format using the
81/// `par_to_far` function.
82///
83/// # Arguments
84/// * `far` - The Fault Address Register value that needs to be translated.
85///
86/// # Returns
87/// * `AxResult<usize>` - The translated HPFAR value, or an error if translation fails.
88///
89/// # Errors
90/// Returns a `BadState` error if the translation is aborted (indicated by the `F` bit in `PAR_EL1`).
91fn translate_far_to_hpfar(far: usize) -> AxResult<usize> {
92    /*
93     * We have
94     *	PAR[PA_Shift - 1 : 12] = PA[PA_Shift - 1 : 12]
95     *	HPFAR[PA_Shift - 9 : 4]  = FIPA[PA_Shift - 1 : 12]
96     */
97    // #define PAR_TO_HPFAR(par) (((par) & GENMASK_ULL(PHYS_MASK_SHIFT - 1, 12)) >> 8)
98    fn par_to_far(par: u64) -> u64 {
99        let mask = ((1 << (52 - 12)) - 1) << 12;
100        (par & mask) >> 8
101    }
102
103    let par = PAR_EL1.get();
104    arm_at!("s1e1r", far);
105    let tmp = PAR_EL1.get();
106    PAR_EL1.set(par);
107    if (tmp & PAR_EL1::F::TranslationAborted.value) != 0 {
108        ax_err!(BadState, "PAR_EL1::F::TranslationAborted value")
109    } else {
110        Ok(par_to_far(tmp) as usize)
111    }
112}

114/// Retrieves the fault address that caused an exception.
115///
116/// This function returns the Guest Physical Address (GPA) that caused the
117/// exception. The address is determined based on the `FAR_EL2` and `HPFAR_EL2`
118/// registers. If the exception is not due to a permission fault or if stage 1
119/// translation is involved, the function uses `HPFAR_EL2` to compute the final
120/// address.
121///
122/// - `far` is the Fault Address Register (FAR_EL2) value.
123/// - `hpfar` is the Hypervisor Fault Address Register (HPFAR_EL2) value,
124///   which might be derived from `FAR_EL2` if certain conditions are met.
125///
126/// The final address returned is computed by combining the page offset from
127/// `FAR_EL2` with the page number from `HPFAR_EL2`.
128///
129/// # Returns
130/// * `AxResult<GuestPhysAddr>` - The guest physical address that caused the exception, wrapped in an `AxResult`.
131#[inline(always)]
132pub fn exception_fault_addr() -> AxResult<GuestPhysAddr> {
133    let far = FAR_EL2.get() as usize;
134    let hpfar =
135        if (exception_esr() & ESR_ELx_S1PTW) == 0 && exception_data_abort_is_permission_fault() {
136            translate_far_to_hpfar(far)?
137        } else {
138            exception_hpfar()
139        };
140    Ok(GuestPhysAddr::from((far & 0xfff) | (hpfar << 8)))
141}

143/// Determines the instruction length based on the ESR_EL2 register.
144///
145/// # Returns
146/// - `1` if the instruction is 32-bit.
147/// - `0` if the instruction is 16-bit.
148#[inline(always)]
149fn exception_instruction_length() -> usize {
150    (exception_esr() >> 25) & 1
151}

153/// Calculates the step size to the next instruction after an exception.
154///
155/// # Returns
156/// The step size to the next instruction:
157/// - `4` for a 32-bit instruction.
158/// - `2` for a 16-bit instruction.
159#[inline(always)]
160pub fn exception_next_instruction_step() -> usize {
161    2 + 2 * exception_instruction_length()
162}

173#[inline(always)]
174pub fn exception_sysreg_direction_write(iss: u64) -> bool {
175    const ESR_ISS_SYSREG_DIRECTION: u64 = 0b1;
176    (iss & ESR_ISS_SYSREG_DIRECTION) == 0
177}

179#[inline(always)]
180pub fn exception_sysreg_gpr(iss: u64) -> u64 {
181    const ESR_ISS_SYSREG_REG_OFF: u64 = 5;
182    const ESR_ISS_SYSREG_REG_LEN: u64 = 5;
183    const ESR_ISS_SYSREG_REG_MASK: u64 = (1 << ESR_ISS_SYSREG_REG_LEN) - 1;
184    (iss >> ESR_ISS_SYSREG_REG_OFF) & ESR_ISS_SYSREG_REG_MASK
185}

187/// The numbering of `SystemReg` follows the order specified in the Instruction Set Specification (ISS),
188/// formatted as `<op0><op2><op1><CRn>00000<CRm>0`.
189/// (Op0[21..20] + Op2[19..17] + Op1[16..14] + CRn[13..10]) + CRm[4..1]
190#[inline(always)]
191pub const fn exception_sysreg_addr(iss: usize) -> usize {
192    const ESR_ISS_SYSREG_ADDR: usize = (0xfff << 10) | (0xf << 1);
193    iss & ESR_ISS_SYSREG_ADDR
194}

196/// Checks if the data abort exception was caused by a permission fault.
197///
198/// # Returns
199/// - `true` if the exception was caused by a permission fault.
200/// - `false` otherwise.
201#[inline(always)]
202pub fn exception_data_abort_is_permission_fault() -> bool {
203    (exception_iss() & 0b111111 & (0xf << 2)) == 12
204}

266/// Macro to save the host function context to the stack.
267///
268/// This macro saves the values of the callee-saved registers (`x19` to `x30`) to the stack.
269/// The stack pointer (`sp`) is adjusted accordingly
270/// to make space for the saved registers.
271///
272/// ## Note
273///
274/// This macro should be used in conjunction with `restore_regs_from_stack!` to ensure that
275/// the saved registers are properly restored when needed,
276/// and the control flow can be returned to `Aarch64VCpu.run()` in `vcpu.rs` happily.
277macro_rules! save_regs_to_stack {
278    () => {
279        "
280        sub     sp, sp, 12 * 8
281        stp     x29, x30, [sp, 10 * 8]
282        stp     x27, x28, [sp, 8 * 8]
283        stp     x25, x26, [sp, 6 * 8]
284        stp     x23, x24, [sp, 4 * 8]
285        stp     x21, x22, [sp, 2 * 8]
286        stp     x19, x20, [sp]"
287    };
288}

290/// Macro to restore the host function context from the stack.
291///
292/// This macro restores the values of the callee-saved general-purpose registers (`x19` to `x30`) from the stack.
293/// The stack pointer (`sp`) is adjusted back after restoring the registers.
294///
295/// ## Note
296///
297/// This macro is called in `return_run_guest()` in exception.rs,
298/// it should only be used after `save_regs_to_stack!` to correctly restore the control flow of `Aarch64VCpu.run()`.
299macro_rules! restore_regs_from_stack {
300    () => {
301        "
302        ldp     x19, x20, [sp]
303        ldp     x21, x22, [sp, 2 * 8]
304        ldp     x23, x24, [sp, 4 * 8]
305        ldp     x25, x26, [sp, 6 * 8]
306        ldp     x27, x28, [sp, 8 * 8]
307        ldp     x29, x30, [sp, 10 * 8]
308        add     sp, sp, 12 * 8"
309    };
